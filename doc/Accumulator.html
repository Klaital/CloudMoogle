def add(datum)
  if (datum.kind_of?(Array))
    # Recurse! Add each element in turn if an Array was passed in
    datum.each {|d| self.add(d)}
  else
    @count += 1
    @min = datum if (datum < @min)
    @max = datum if (datum > @max)
    @sum += datum
    @mean = @sum / @count.to_f
  end
end
