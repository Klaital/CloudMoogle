def d(msg=nil)
  s = if (block_given? && CONFIGS[:logs][:level] <= 0)
    yield
  else
    msg
  end

  puts "DEBUG #{s}" if (CONFIGS[:logs][:stdout] && CONFIGS[:logs][:level] <= 0)
  debug s
end

def metric( data = {} )
  data = data.collect do |k,v| 
    val = (v.respond_to?(:iso8601)) ? v.iso8601 : v.to_s
    "#{k}=#{val}\n"
  end
  debug "#{SPACER}\n#{data.join('')}"
end
