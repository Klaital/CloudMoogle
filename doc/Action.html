def initialize
  @data = {
    :type => nil,
    :subtype => nil,
    :actor => nil,
    :target => nil,
    :damage => nil, # nil damage on a combat action implies a miss
    :ability_name => nil,
    :light => nil,
    :format => nil
  }
  @incomplete = true
end

def pattern_name
  return nil if (@data[:type].nil? || @data[:subtype].nil?)
  "#{@data[:type].downcase}_#{@data[:subtype].downcase}"
end

def to_json
  j = "{\n"
  @data.each_pair {|k,v| j += "#{k}:'#{v}'\n"}
  j += '}'
end
