    :target => nil,
    :damage => nil, # nil damage on a combat action implies a miss

def Action.parse_tsv(line)
  a = Action.new
  tokens = line.split("\t").collect {|t| (t.strip.length == 0) ? nil : t.strip}
  return nil if (tokens.length < 8)
  a.format = tokens[0]
  a.type = tokens[1]
  a.subtype = tokens[2]
  a.actor = tokens[3]
  a.target = tokens[4]
  a.damage = tokens[5]
  a.ability_name = tokens[6]
  a.light = tokens[7]
  
  return a
end
